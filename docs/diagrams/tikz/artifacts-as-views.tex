% Artifacts as views. One claim: a card can only say what the store can back.
% Card, Atlas and Safety case are lenses over an append-only evidence store; each
% cites nodes it does not compute. A manuscript number tagged with an evidence id
% that is not in the store fails the claims-check and exits nonzero.

% a biconvex lens: the artifact-as-view glyph. #1 cx #2 cy #3 half-width #4 label
\newcommand{\rllens}[4]{%
  \draw[draw=rlAccent, line width=1.0pt, fill=rlAccent, fill opacity=0.08]
    (#1-#3,#2)
      .. controls (#1-#3*0.5,#2+0.62) and (#1+#3*0.5,#2+0.62) .. (#1+#3,#2)
      .. controls (#1+#3*0.5,#2-0.62) and (#1-#3*0.5,#2-0.62) .. (#1-#3,#2) -- cycle;
  \node[font=\sffamily\bfseries\small, text=rlAccent] at (#1,#2) {#4};%
}
